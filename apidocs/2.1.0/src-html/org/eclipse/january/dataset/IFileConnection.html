010package org.eclipse.january.dataset;
011
012public interface IFileConnection {
013
014        /**
015         * DataServer path, a local path on the server used to locate the remote dataset.
016         * @return the file path to the data in the file system of the remote machine
017         */
018        public String getPath();
019        
020        /**
021         * DataServer path, a local path on the server used to locate the remote dataset.
022         * This path may also be a directory where the data collection will happen. In this
023         * case when the first file is written, the dataset must be made up of files with the
024         * same extension.
025         * 
026         * @param path to the data in the file system of the remote machine
027         */
028        public void setPath(String path);
029
030}
