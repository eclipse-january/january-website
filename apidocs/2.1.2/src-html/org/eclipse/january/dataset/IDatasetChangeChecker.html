010package org.eclipse.january.dataset;
011
012/**
013 * Interface for a dataset checker to detect change
014 */
015public interface IDatasetChangeChecker {
016
017        /**
018         * Set dataset to be checked for signs of change. Do any state initialization in here.
019         * @param dataset
020         */
021        public void setDataset(ILazyDataset dataset);
022
023        /**
024         * Check for any change. Also update state information.
025         * @return true if possible change may have occurred
026         */
027        public boolean check();
028}
